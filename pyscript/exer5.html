import json import pyodide from js import Bokeh, console, JSON from bokeh.embed import json_item from bokeh.plotting import figure from bokeh.resources import CDN vegetables = ['Carrot', 'Onion', 'Potato', 'Cabbage', 'garlic'] values = [15, 23, 34, 14, 12] pl = figure(x_range=vegetables, height=350, title="Vegetables Counts", toolbar_location=None, tools="") pl.vbar(x=vegetables, top=values, width=0.9) pl.xgrid.grid_line_color = None pl.y_range.start = 0 p_json = json.dumps(json_item(pl, "chart")) Bokeh.embed.embed_item(JSON.parse(p_json))
